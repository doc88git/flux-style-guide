<template>
  <div class="f-chip" :class="classes" @click="onClick">
    <div class="f-chip__icon" v-if="icon">
      <slot name="icon">
        <f-icon :name="selected ? 'check' : icon" />
      </slot>
    </div>
    <div class="f-chip__content">
      <slot>{{ label }}</slot>
    </div>
    <div
      class="f-chip__close"
      v-if="removable"
      @click="onRemove($event, value)"
    >
      <slot name="close">
        <f-icon name="close" size="sm" />
      </slot>
    </div>
  </div>
</template>
 
<script>
import { FIcon } from '..'
export default {
  name: 'f-chip',
  components: {
    FIcon
  },
  props: {
    label: String,
    value: [String, Number],
    color: {
      type: String,
      default: 'primary'
    },
    textColor: String,
    icon: String,
    removable: Boolean,
    disable: Boolean,
    selected: {
      type: Boolean,
      default: null
    }
  },
  computed: {
    colors() {
      return {
        [`color--text--${this.textColor}`]: !!this.textColor,
        [`color--${this.color}`]: !!this.color
      }
    },
    isDisabled() {
      if (!this.disable) return {}
      return { ['f-chip--disabled']: this.disable }
    },
    classes() {
      return {
        ...this.colors,
        ...this.isDisabled
      }
    }
  },
  methods: {
    onRemove(e, value) {
      if (!this.disable) this.$emit('remove', value)
 
      e.stopPropagation()
    },
    onClick(e) {
      if (this.disable) return false
 
      this.$emit('update:selected', !this.selected)
      this.$emit('click', e)
 
      e.stopPropagation()
    }
  }
}
</script>
